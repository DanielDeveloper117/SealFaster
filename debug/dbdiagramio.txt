  proveedor varchar(255)
  material varchar(255)
  Clave varchar(255)
  Medida varchar(255)
  max_usable int
  pre_stock decimal(10,2)
  stock decimal(10,2)
  lote_pedimento varchar(255)
  precio decimal(10,2)
  interior int
  exterior int
  estatus varchar(50)
  solicita_archivado tynint(2)
  justificacion_archi varchar(255)
  ruta_foto_barra varchar(255)
  archivado_auth tynint(2)
  created_at datetime
  updated_at datetime
  delated_at datetime

  indexes {

  }
}

Table almacenes {
  id bigint [pk, increment]
  almacen varchar(50) [not null]

  updated_at timestamp
  created_at timestamp [default: `now()`]

  indexes {

    almacen_origen_id
    almacen_destino_id
    recibio_id
  }
}
Table familias {
  id bigint [pk, increment]
  nombre varchar(30)
  nombre2 varchar(30)
  descripcion varchar(255)
  updated_at timestamp
  created_at timestamp [default: `now()`]
}

Table perfiles2 {
  id bigint [pk, increment]
  familia_id bigint [ref: > familias.id]
  grupo_herramienta_id bigint [ref: > grupos_herramienta.id, null]
  nombre varchar(30)
  detalles varchar(255)
  cantidad_componentes smallint
  con_resorte_en tinyint [default: 0]
  es_wiper_en tinyint [default: 0]
  con_escalon_en tinyint [default: 0]
  es_wiper_especial_en tinyint [default: 0]
  updated_at timestamp
  created_at timestamp [default: `now()`]

  indexes {
    familia_id
    grupo_herramienta_id
  }
}

Table porcentajes_perfil {
  id bigint [pk, increment]
  perfil_id bigint [ref: > perfiles2.id]
  componente smallint
  tipo enum('DI', 'DE', 'H') [not null]
  porcentaje decimal(6,3) [default: 1]
  updated_at timestamp
  created_at timestamp [default: `now()`]

  indexes {
    perfil_id
    (perfil_id, componente, tipo) [unique]
    // un componente no puede tener dos porcentajes del mismo tipo
  }
}
Table tolerancias_perfil {
  id bigint [pk, increment]
  perfil_id bigint [ref: > perfiles2.id]
  componente smallint
  tipo enum('DI', 'DE') [not null]
  // Solo DI y DE, la altura no tiene tolerancia de barra
  tolerancia decimal(6,2) [not null]
  updated_at timestamp
  created_at timestamp [default: `now()`]

  indexes {
    perfil_id
    (perfil_id, componente, tipo) [unique]
    // un componente no puede tener dos tolerancias del mismo tipo
  }
}

Table grupos_herramienta {
  // Agrupa perfiles que comparten el mismo conjunto de limitantes de herramienta.
  // Equivale a limitantesHerramientas1..10 en el codigo original.
  // Ejemplo: grupo "estandar", grupo "doble_sello", grupo "o_ring", etc.
  id bigint [pk, increment]
  nombre varchar(60)
  descripcion varchar(255)
  updated_at timestamp
  created_at timestamp [default: `now()`]
}

Table herramientas {
  // Catalogo de herramientas CNC disponibles.
  // Cada herramienta tiene un numero identificador unico (112, 212, 103, etc.)
  id bigint [pk, increment]
  numero varchar(40)
  // varchar y no int porque existen claves como "h023", "h024"
  descripcion varchar(255)
  updated_at timestamp
  created_at timestamp [default: `now()`]
}

Table limitantes_herramienta {
  // Define los rangos validos de dimension para una herramienta especifica
  // dentro de un grupo de perfiles y para una dureza determinada.
  // Una herramienta puede aparecer multiples veces: una vez por grupo
  // y una vez por dureza dentro de ese grupo.
  id bigint [pk, increment]
  grupo_herramienta_id bigint [ref: > grupos_herramienta.id]
  herramienta_id bigint [ref: > herramientas.id]
  dureza enum('blandos', 'duros') [not null]
  di_min decimal(7,2)
  di_max decimal(7,2)
  de_min decimal(7,2)
  de_max decimal(7,2)
  seccion_min decimal(7,2)
  seccion_max decimal(7,2)
  h_min decimal(7,2)
  h_max decimal(7,2)
  updated_at timestamp
  created_at timestamp [default: `now()`]

  indexes {
    grupo_herramienta_id
    herramienta_id
    (grupo_herramienta_id, herramienta_id, dureza) [unique]
    // unicidad: una herramienta no puede tener dos filas para la misma
    // combinacion de grupo y dureza
  }
}

Table claves_alternas {

  created_at timestamp [default: `now()`]
}

  billets_claves_lotes varchar(1800)
  billets_string varchar(2000)
  billets_string2 varchar(2300)
  billets_manualmente varchar(600)
  
  // Dimensiones Técnicas (Métricas)
  tipo_medida varchar(10) [default: "Sello"]
  altura decimal(10,2) [default: 0.00]
  altura_caja decimal(10,2) [default: 0.00]
  altura_escalon decimal(10,2) [default: 0.00]
  altura_h2 decimal(10,2) [default: 0.00]
  altura_h3 decimal(10,2) [default: 0.00]
  diametro_int decimal(10,2) [default: 0.00]
  diametro_ext decimal(10,2) [default: 0.00]
  a_sello decimal(10,2) [not null, default: 0.00]
  di_sello decimal(10,2) [not null, default: 0.00]
  de_sello decimal(10,2) [not null, default: 0.00]
  
  // Dimensiones Técnicas (Pulgadas / Secundarias)
  a_sello_inch decimal(10,4) [not null, default: 0.0000]
  di_sello_inch decimal(10,4) [not null, default: 0.0000]
  de_sello_inch decimal(10,4) [not null, default: 0.0000]
  a_sello2 decimal(10,2) [not null, default: 0.00]
  di_sello2 decimal(10,2) [not null, default: 0.00]
  de_sello2 decimal(10,2) [not null, default: 0.00]
  
  // Totales y Descuentos

  total_unitarios decimal(10,2)
  desc_cliente decimal(10,2)
  desc_cantidad decimal(10,2)
  desc_mayoreo decimal(10,2)

  codigo_cliente varchar(20)
  correo_cliente varchar(100)
  vendedor varchar(255)

  img varchar(255)

  hora time [default: `curtime()`]
  fecha_actualizacion datetime
  fecha_insercion datetime
  fecha_vencimiento datetime
  archivada tinyint(4) [default: 0]
  simulacion tinyint(2) [not null, default: 0]

  // Índices para optimización de performance
  indexes {
    (id_fusion, id_cotizacion, id_estimacion) [name: "idx_performance_optim"]
    (fecha, id_usuario, archivada) [name: "idx_filtros_rapidos"]
  }
}

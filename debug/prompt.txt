SELECT 
    id,
    clave,
    clave_alterna,
    material,
    -- Creamos una columna de diagnóstico para saber qué campo está sucio
    CASE 
        WHEN clave LIKE CONCAT('%', CHAR(0), '%') THEN 'Clave con Null Byte'
        WHEN clave <> TRIM(clave) THEN 'Clave con espacios'
        WHEN material LIKE CONCAT('%', CHAR(0), '%') THEN 'Material con Null Byte'
        WHEN clave_alterna LIKE CONCAT('%', CHAR(0), '%') THEN 'Clave Alterna con Null Byte'
        ELSE 'Otro campo (Proveedor/Tipo)'
    END AS diagnostico_error
FROM parametros
WHERE 
    -- Detectar CHAR(0) en cualquier columna
    clave         LIKE CONCAT('%', CHAR(0), '%') OR
    clave_alterna LIKE CONCAT('%', CHAR(0), '%') OR
    material      LIKE CONCAT('%', CHAR(0), '%') OR
    proveedor     LIKE CONCAT('%', CHAR(0), '%') OR
    tipo          LIKE CONCAT('%', CHAR(0), '%') OR
    -- Detectar espacios al inicio o final (que TRIM eliminaría)
    clave         <> TRIM(clave) OR
    clave_alterna <> TRIM(clave_alterna) OR
    material      <> TRIM(material)
ORDER BY id ASC;

UPDATE parametros
SET
    clave         = TRIM(REPLACE(clave,         CHAR(0), '')),

Table inventario_cnc {
  id int [pk, increment]
  almacen_id bigint
  operacion_id bigint
  proveedor varchar(255)
  material varchar(255)
  Clave varchar(255)
  Medida varchar(255)
  max_usable int
  pre_stock decimal(10,2)
  stock decimal(10,2)
  lote_pedimento varchar(255)
  precio decimal(10,2)
  interior int
  exterior int
  estatus varchar(50)
  solicita_archivado tynint(2)
  justificacion_archi varchar(255)
  ruta_foto_barra varchar(255)
  archivado_auth tynint(2)
  created_at datetime
  updated_at datetime
  delated_at datetime

  indexes {
    almacen_id
  }
}

Table almacenes {
  id bigint [pk, increment]
  almacen varchar(50) [not null]
  descripcion varchar(255)
  updated_at timestamp
  created_at timestamp [default: `now()`]
}

Table operaciones_inv {
  id bigint [pk, increment]
  usuario_id bigint
  tipo enum('Traspaso','Venta') [not null]
  almacen_origen_id bigint [not null]
  almacen_destino_id bigint [not null]
  justificacion varchar(255) [not null]
  img_envio_barras varchar(255)
  img_envio_paquete varchar(255)
  img_recepcion_paquete varchar(255)
  img_recepcion_barras varchar(255)
  recibido tinyint  [default: 0]
  recibio_id bigint 
  fecha_recibido timestamp
  updated_at timestamp
  created_at timestamp [default: `now()`]

  indexes {
    usuario_id
    almacen_origen_id
    almacen_destino_id
    recibio_id
  }
}

Ref: inventario_cnc.almacen_id > almacenes.id
Ref: operaciones_inv.almacen_origen_id > almacenes.id
Ref: operaciones_inv.almacen_destino_id > almacenes.id
Ref: inventario_cnc.operacion_id > operaciones_inv.id
